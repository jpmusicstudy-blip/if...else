#include <stdio.h>
#include <string.h>

int main() {
    char senha[20];

    printf("Digite a senha: ");
    fgets(senha, 20, stdin);

    // Remove o ENTER (\n)
    senha[strcspn(senha, "\n")] = '\0';

    if (strcmp(senha, "1234") == 0) {
        printf("Acesso permitido\n");
    } else {
        printf("Acesso negado\n");
    }

    return 0;
}
