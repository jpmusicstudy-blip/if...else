#include <stdio.h>
#include <string.h>

int main() {
    char senha[20];
    int tentativas = 0;

    while (tentativas < 3) {
        printf("Digite a senha: ");
        scanf("%s", senha);

        if (strcmp(senha, "1234") == 0) {
            printf("Acesso permitido\n");
            break;
        } else {
            printf("Senha incorreta\n");
        }

        tentativas++;
    }

    if (tentativas == 3) {
        printf("Acesso bloqueado\n");
    }

    return 0;
}
